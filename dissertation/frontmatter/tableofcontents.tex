% Mathematical Notation
\chapter*{Mathematical Notation}
\addcontentsline{toc}{chapter}{Mathematical Notation}

\begin{description}[style=nextline,leftmargin=3cm,labelsep=1em]
\item[$\agents$] Set of all agents in the model
\item[$\agent{i}$] Individual agent $i$
\item[$\network{G}$] Social network represented as a graph
\item[$\adjmatrix{X}$] Adjacency matrix of the network
\item[$\parameters$] Vector of model parameters
\item[$\parameter{\theta}$] Individual parameter $\theta$
\item[$\estimate{\theta}$] Estimated value of parameter $\theta$
\item[$\prob{A}$] Probability of event $A$
\item[$\expect{X}$] Expected value of random variable $X$
\item[$\var{X}$] Variance of random variable $X$
\item[$\cov{X,Y}$] Covariance between variables $X$ and $Y$
\item[$\likelihood{\theta}$] Likelihood function for parameters $\theta$
\item[$\loglik{\theta}$] Log-likelihood function for parameters $\theta$
\item[$\posterior{\theta}$] Posterior distribution of parameters $\theta$
\item[$\timestep{t}$] Time step $t$ in discrete time models
\item[$\timeinterval{s}{t}$] Time interval from $s$ to $t$ in continuous time models
\item[$\indicator{A}$] Indicator function that equals 1 if condition $A$ is true, 0 otherwise
\item[$\sigmoid{x}$] Sigmoid function $\frac{1}{1+e^{-x}}$
\item[$\degree{i}$] Degree centrality of node $i$
\item[$\clustering{i}$] Clustering coefficient of node $i$
\item[$\centrality{measure}{i}$] Centrality measure for node $i$
\end{description}
